Code Example: Form Validation

This is a snippet of JavaScript used in my newsletter signup form to validate user input before submission.


            // Display welcome alert on page load
            function welcomeMessage() {
              alert("Welcome to XYZ Corporation - Newsletter Signup");
            }

            // Validate first and last name fields
            function validateForm() {
              const firstName = document.getElementById("firstName").value.trim();
              const lastName = document.getElementById("lastName").value.trim();

              if (!firstName || !lastName) {
                alert("Please fill in both your First and Last Name.");
                return false;
              }
              return true;
            }
